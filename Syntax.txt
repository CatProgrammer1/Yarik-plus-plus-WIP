yar - кейворд для создания переменных
Пример:
yar a = 1

fn - кейворд для создания функций
Пример:
fn a() {
  println("Yarik++ хохохо")
}

wl - кейворд обьявления цикла while
Пример:
wl true == true {
  println("Yarik++")
}

if - кейворд для логических условий(Пока что нету else, else if)
Пример:
if true == true {
    println("Тру")
}

&& - и
Пример:
yar a = 1 == 1 && 0 == 0

?? - или
Пример:
yar a = 1 == 0 ?? 0 == 0

==, !=, >, <, >=, <= - логические операторы
Пример:
yar a = 1

yar isGreater = a > 5 #-> фолссс

+, -, *, /, %, ^ - арифметические операторы
Пример:
yar x = 5
yar y = x * 2

println(y) #-> 10

# - коментарии
Пример:
yar a = 1 #Создаём переменную a с значением 1
